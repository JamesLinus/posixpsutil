def initialize(opt={})
  raise ArgumentError if opt[:pid].nil?
  @pid = opt[:pid] # pid must given
  @name = opt[:name] || nil
  if opt[:msg].nil?
    if @name
      details = "(pid=#{@pid}, name=#{@name.to_s})"
    else
      details = "(pid=#{@pid})"
    end
    opt[:msg] = "process no longer exists " + details
  end
  @message = opt[:msg]
end
